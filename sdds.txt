package RECAP1;

import java.util.Scanner;

public class main {

	public static void main(String[] args) {
		
		Scanner scan = new Scanner(System.in);
		
		String Poli = scan.next();
		
		Poli = isPalindrom(Poli);
			System.out.println(Poli);
		}
		

		

	
	public static String isPalindrom(String Poli) {
		String str= "";
		for (int j = 0 , i = Poli.length(); i >= 0; i++, i--) {
			if (Poli.charAt(i) != Poli.charAt(j)) {
				str = "Kein Poli";
				
			}
			else if (Poli.charAt(i) == Poli.charAt(j)) {
				str = Poli;
				
			
			}
	}
		return str;
	}
}
